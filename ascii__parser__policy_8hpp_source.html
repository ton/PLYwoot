49class AsciiParserPolicy
50{
51public:
52 AsciiParserPolicy(std::istream &is) : is_{is} {}
53
56 void skipElement(const PlyElement &e) const { is_.skipLines(e.size()); }
57
59 void skipProperty(const PlyProperty &p) const
60 {
61 if (p.isList())
62 {
63 // TODO(ton): maybe switch on the list size type...?
64 const auto size = readNumber<int>();
65 for (int i = 0; i < size; ++i) { skipNumber(); }
66 }
67 else { skipNumber(); }
68 }
69
71 template<typename T>
72 T readNumber() const
73 {
74 is_.skipWhitespace();
75 if (is_.eof()) { throw UnexpectedEof(); }
76
77 is_.buffer(256);
78 return detail::to_number<T>(is_.data(), is_.data() + 256, &is_.data());
79 }
80
85 template<typename PlyT, typename DestT, std::size_t N>
86 std::uint8_t *readNumbers(std::uint8_t *dest) const
87 {
88 // TODO(ton): needs to be specialized for improved performance.
89 for (std::size_t i = 0; i < N; ++i, dest += sizeof(DestT))
90 {
91 *reinterpret_cast<DestT *>(dest) = readNumber<PlyT>();
92 }
93 return dest;
94 }
95
97 template<typename T = void>
98 void skipNumber() const
99 {
100 is_.skipWhitespace();
101 is_.skipNonWhitespace();
102 }
103
105 void skipProperties(size_t n) const
106 {
107 if (n > 0) is_.skipLines(1);
108 }
109
110private:
111 mutable detail::BufferedIStream is_;
112};

107namespace plywoot::detail {
108
111class HeaderParser
112{
113 using Token = detail::HeaderScanner::Token;
114
115public:
120 HeaderParser(std::istream &is) : scanner_{is}
121 {
122 accept(Token::MagicNumber);
123
124 // Parse the format section.
125 accept(Token::Format);
126 switch (scanner_.nextToken())
127 {
128 case Token::Ascii:
129 format_ = PlyFormat::Ascii;
130 break;
131 case Token::BinaryLittleEndian:
132 format_ = PlyFormat::BinaryLittleEndian;
133 break;
134 case Token::BinaryBigEndian:
135 format_ = PlyFormat::BinaryBigEndian;
136 break;
137 default:
138 throw InvalidFormat(scanner_.tokenString());
139 }
140 scanner_.nextToken(); // ignore the format version
141
142 // Ignore comment section for now.
143 while (scanner_.nextToken() == Token::Comment) { comments_.push_back(scanner_.comment()); }
144
145 // Parse elements.
146 do {
147 switch (scanner_.token())
148 {
149 case Token::EndHeader:
150 break;
151 case Token::Element:
152 elements_.push_back(parseElement());
153 break;
154 case Token::Comment:
155 comments_.push_back(scanner_.comment());
156 scanner_.nextToken();
157 break;
158 default:
159 throw UnexpectedToken(scanner_.token(), scanner_.tokenString());
160 break;
161 }
162 } while (scanner_.token() != Token::EndHeader);
163 }
164
169 const std::vector<Comment> &comments() const { return comments_; }
173 const std::vector<PlyElement> &elements() const { return elements_; }
174
178 PlyFormat format() const { return format_; }
179
180private:
183 void accept(Token expected)
184 {
185 if (scanner_.nextToken() != expected)
186 {
187 // In case an identifier token is expected, all reserved keywords are
188 // acceptable as well.
189 if (!(expected == Token::Identifier && scanner_.isKeyword(scanner_.token())))
190 {
191 throw UnexpectedToken(expected, scanner_.token(), scanner_.tokenString());
192 }
193 }
194 }
195
198 PlyDataType tokenToDataType(Token t) const
199 {
200 switch (t)
201 {
202 case Token::Char:
203 return PlyDataType::Char;
204 case Token::UChar:
205 return PlyDataType::UChar;
206 case Token::Short:
207 return PlyDataType::Short;
208 case Token::UShort:
209 return PlyDataType::UShort;
210 case Token::Int:
211 return PlyDataType::Int;
212 case Token::UInt:
213 return PlyDataType::UInt;
214 case Token::Float:
215 return PlyDataType::Float;
216 case Token::Double:
217 return PlyDataType::Double;
218 default:
219 // TODO(ton): expected is a range of tokens...
220 throw UnexpectedToken(Token::Char, t, scanner_.tokenString());
221 }
222 }
223
225 PlyElement parseElement()
226 {
227 accept(Token::Identifier); // name of the elements
228 std::string name{scanner_.tokenString()};
229
230 accept(Token::Number);
231 std::size_t size{scanner_.tokenNumber()};
232
233 PlyElement result{std::move(name), size};
234
235 // Parse properties.
236 while (scanner_.nextToken() == Token::Property || scanner_.token() == Token::Comment)
237 {
238 if (scanner_.token() == Token::Property)
239 {
240 PlyDataType type;
241 PlyDataType sizeType;
242
243 // TODO(ton): probably reserved keywords may be used as names as well,
244 // just accept every token here, even numbers?
245 switch (scanner_.nextToken())
246 {
247 case Token::List:
248 sizeType = tokenToDataType(scanner_.nextToken());
249 type = tokenToDataType(scanner_.nextToken());
250 accept(Token::Identifier);
251 result.addProperty(scanner_.tokenString(), type, sizeType);
252 break;
253 default:
254 type = tokenToDataType(scanner_.token());
255 accept(Token::Identifier);
256 result.addProperty(scanner_.tokenString(), type);
257 break;
258 }
259 }
260 else
261 {
262 assert(scanner_.token() == Token::Comment);
263 comments_.push_back(scanner_.comment());
264 }
265 }
266
267 return result;
268 }
269
271 std::vector<Comment> comments_;
273 PlyFormat format_;
275 std::vector<PlyElement> elements_;
277 detail::HeaderScanner scanner_;
278};

53 void write(const PlyElement &element, const std::uint8_t *src, std::size_t alignment) const
54 {
55 for (std::size_t i = 0; i < element.size(); ++i)
56 {
57 const std::vector<PlyProperty> &properties = element.properties();
58 const auto first = properties.begin();
59 const auto last = properties.end();
60
61 auto curr = first;
62 while (curr < last)
63 {
64 if (curr > first) { this->writeTokenSeparator(); }
65
66 const PlyProperty &property = *curr++;
67 if (property.isList())
68 {
69 switch (property.type())
70 {
71 case PlyDataType::Char:
72 src = writeProperty(src, property, reflect::Type<std::vector<char>>{});
73 break;
74 case PlyDataType::UChar:
75 src = writeProperty(src, property, reflect::Type<std::vector<unsigned char>>{});
76 break;
77 case PlyDataType::Short:
78 src = writeProperty(src, property, reflect::Type<std::vector<short>>{});
79 break;
80 case PlyDataType::UShort:
81 src = writeProperty(src, property, reflect::Type<std::vector<unsigned short>>{});
82 break;
83 case PlyDataType::Int:
84 src = writeProperty(src, property, reflect::Type<std::vector<int>>{});
85 break;
86 case PlyDataType::UInt:
87 src = writeProperty(src, property, reflect::Type<std::vector<unsigned int>>{});
88 break;
89 case PlyDataType::Float:
90 src = writeProperty(src, property, reflect::Type<std::vector<float>>{});
91 break;
92 case PlyDataType::Double:
93 src = writeProperty(src, property, reflect::Type<std::vector<double>>{});
94 break;
95 }
96 }
97 else
98 {
99 switch (property.type())
100 {
101 case PlyDataType::Char:
102 src = writeProperty(src, property, reflect::Type<char>{});
103 break;
104 case PlyDataType::UChar:
105 src = writeProperty(src, property, reflect::Type<unsigned char>{});
106 break;
107 case PlyDataType::Short:
108 src = writeProperty(src, property, reflect::Type<short>{});
109 break;
110 case PlyDataType::UShort:
111 src = writeProperty(src, property, reflect::Type<unsigned short>{});
112 break;
113 case PlyDataType::Int:
114 src = writeProperty(src, property, reflect::Type<int>{});
115 break;
116 case PlyDataType::UInt:
117 src = writeProperty(src, property, reflect::Type<unsigned int>{});
118 break;
119 case PlyDataType::Float:
120 src = writeProperty(src, property, reflect::Type<float>{});
121 break;
122 case PlyDataType::Double:
123 src = writeProperty(src, property, reflect::Type<double>{});
124 break;
125 }
126 }
127 }
128
129 src = detail::align(src, alignment);
130
131 this->writeNewline();
132 }
133 }

135 template<typename... Ts>
136 void write(const PlyElement &element, const reflect::Layout<Ts...> layout) const
137 {
138 const auto first = element.properties().begin();
139 const auto last = element.properties().end();
140
141 const std::uint8_t *src = layout.data();
142 for (std::size_t i = 0; i < layout.size(); ++i)
143 {
144 src = detail::align(writeElement<Ts...>(src, first, last), layout.alignment());
145 }
146 }

149 template<typename PlyT, typename SrcT>
150 const std::uint8_t *writeProperty(const std::uint8_t *src, reflect::Type<SrcT>) const
151 {
152 if constexpr (std::is_arithmetic_v<SrcT>)
153 {
154 src = static_cast<const std::uint8_t *>(detail::align(src, alignof(SrcT)));
155 this->writeNumber(static_cast<PlyT>(*reinterpret_cast<const SrcT *>(src)));
156 return src + sizeof(SrcT);
157 }
158 else { return static_cast<const std::uint8_t *>(detail::align(src, alignof(SrcT))) + sizeof(SrcT); }
159 }

163 template<typename PlyT, typename SrcT, std::size_t N>
164 const std::uint8_t *writeProperty(const std::uint8_t *src, reflect::Type<reflect::Pack<SrcT, N>>) const
165 {
166 static_assert(N > 0, "invalid pack size specified (needs to be larger than zero)");
167 static_assert(
168 std::is_arithmetic<SrcT>::value, "it is not (yet) possible to write packs of non-numeric types");
169
170 src = static_cast<const std::uint8_t *>(detail::align(src, alignof(SrcT)));
171 this->template writeNumbers<PlyT, SrcT>(reinterpret_cast<const SrcT *>(src), N);
172 return src + N * sizeof(SrcT);

311 template<typename Policy, typename T, typename U, typename... Ts>
312 const std::uint8_t *writeProperties(
313 const std::uint8_t *src,
314 PropertyConstIterator first,
315 PropertyConstIterator last) const
316 {
317 if constexpr (std::is_same_v<Policy, AsciiWriterPolicy>)
318 {
319 src = writeProperty<T>(src, first, last);
320 first += detail::numProperties<T>();
321 if (first < last) { this->writeTokenSeparator(); }
322 return writeProperties<Policy, U, Ts...>(src, first, last);
323 }
324 else
325 {
326 return writeProperties<Policy, U, Ts...>(
327 writeProperty<T>(src, first, last), first + detail::numProperties<T>(), last);
328 }
329 }
331
332 template<typename... Ts>
333 const std::uint8_t *writeElement(
334 const std::uint8_t *src,
335 PropertyConstIterator first,
336 PropertyConstIterator last) const
337 {
338 src = this->writeProperties<FormatWriterPolicy, Ts...>(src, first, last);
339
340 // In case the element defines more properties than the source data,
341 // append the missing properties with a default value of zero.
342 if (detail::numProperties<Ts...>() < static_cast<std::size_t>(std::distance(first, last)))
343 {
344 this->writeMissingProperties(first + detail::numProperties<Ts...>(), last);
345 }
346
347 this->writeNewline();
348
349 return src;
350 }

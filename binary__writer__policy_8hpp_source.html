31template<typename Endianness>
32class BinaryWriterPolicy
33{
34public:
35 BinaryWriterPolicy(std::ostream &os) : os_{os} {}
36
39 template<typename T, typename EndiannessDependent = Endianness>
40 void writeNumber(T t) const
41 {
42 // TODO(ton): this assumes the target architecture is little endian.
43 if constexpr (std::is_same_v<EndiannessDependent, LittleEndian>)
44 {
45 os_.write(reinterpret_cast<const char *>(&t), sizeof(T));
46 }
47 else
48 {
49 const auto be = htobe(t);
50 os_.write(reinterpret_cast<const char *>(&be), sizeof(T));
51 }
52 }
53
58 template<typename PlySizeT, typename PlyT, typename SrcT, typename EndiannessDependent = Endianness>
59 void writeList(const SrcT *t, std::size_t n) const
60 {
61 writeNumber<PlySizeT>(n);
62 writeNumbers<PlyT, SrcT>(t, n);
63 }
64
67 template<typename PlyT, typename SrcT, typename EndiannessDependent = Endianness>
68 void writeNumbers(const SrcT *t, std::size_t n) const
69 {
70 if constexpr (std::is_same_v<PlyT, SrcT> && std::is_same_v<EndiannessDependent, LittleEndian>)
71 {
72 os_.write(reinterpret_cast<const char *>(t), sizeof(SrcT) * n);
73 }
74 else
75 {
76 for (std::size_t i = 0; i < n; ++i) { writeNumber<PlyT>(*t++); }
77 }
78 }

85 void writeMissingProperties(PlyPropertyConstIterator first, PlyPropertyConstIterator last) const
86 {
87 while (first != last)
88 {
89 switch (first->isList() ? first->sizeType() : first->type())
90 {
91 case PlyDataType::Char:
92 writeNumber<char>(0);
93 break;
94 case PlyDataType::UChar:
95 writeNumber<unsigned char>(0);
96 break;
97 case PlyDataType::Short:
98 writeNumber<short>(0);
99 break;
100 case PlyDataType::UShort:
101 writeNumber<unsigned short>(0);
102 break;
103 case PlyDataType::Int:
104 writeNumber<int>(0);
105 break;
106 case PlyDataType::UInt:
107 writeNumber<unsigned int>(0);
108 break;
109 case PlyDataType::Float:
110 writeNumber<float>(0);
111 break;
112 case PlyDataType::Double:
113 writeNumber<double>(0);
114 break;
115 }
116 ++first;
117 }
118 }

121 void writeNewline() const {}

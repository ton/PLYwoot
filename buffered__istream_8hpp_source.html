63 template<typename T>
64 inline T read()
65 {
66 // Note; buffer a bit more than strictly necessary, so that we can move the
67 // read head unconditionally after reading the object of type T.
68 if (c_ + sizeof(T) > eob_) buffer(sizeof(T));
69 const char *t = c_;
70 c_ += sizeof(T);
71 return *reinterpret_cast<const T *>(t);
72 }

77 inline std::uint8_t *memcpy(std::uint8_t *dest, std::size_t n)
78 {
79 if (n > IStreamBufferSize)
80 {
81 const std::size_t remaining = eob_ - c_;
82 std::memcpy(dest, c_, remaining);
83 is_.read(reinterpret_cast<char *>(dest) + remaining, n - remaining);
84 c_ = eob_;
85 }
86 else
87 {
88 if (c_ + n > eob_) buffer(n);
89
90 std::memcpy(dest, c_, n);
91 c_ += n;
92 }
93
94 return dest + n;
95 }

100 template<typename From, typename To, size_t N>
101 std::uint8_t * read(std::uint8_t *dest)
102 {
103 if constexpr (std::is_same_v<From, To>)
104 {
105 // Little endian; source number type is equal to the destination number
106 // type.
107 return this->memcpy(dest, N * sizeof(From));
108 }
109 else
110 {
111 // TODO(ton): broken; this might overflow in case `bytesToRead >
112 // IStreamBufferSize` holds.
113 constexpr std::size_t bytesToRead = N * sizeof(From);
114 if (c_ + bytesToRead > eob_) buffer(bytesToRead);
115
116 const From *from = reinterpret_cast<const From *>(c_);
117 To *to = reinterpret_cast<To *>(dest);
118 for (size_t i = 0; i < N; ++i) { *to++ = *from++; }
119
120 c_ += bytesToRead;
121
122 return reinterpret_cast<std::uint8_t *>(to);
123 }
124 }

127 void skip(std::size_t n)
128 {
129 const std::size_t remaining = eob_ - c_;
130 if (remaining > n) { c_ += n; }
131 else
132 {
133 is_.seekg(n - remaining, std::ios_base::cur);
134 buffer();
135 }
136 }

141 void skipLines(std::size_t n)
142 {
143 while (*c_ != EOF && n > 0)
144 {
145 c_ = static_cast<const char *>(std::memchr(c_, '\n', eob_ - c_));
146 if (c_)
147 {
148 readCharacter();
149 --n;
150 }
151 else { buffer(); }
152 }
153 }

158 inline void skipWhitespace()
159 {
160 while (0 <= *c_ && *c_ <= 0x20) { readCharacter(); }
161 }

166 inline void skipNonWhitespace()
167 {

175 void buffer(std::size_t minimum)
176 {
177 std::size_t remaining = eob_ - c_;
178 if (remaining < minimum)
179 {
180 std::memcpy(buffer_, c_, remaining);
181 if (!is_.read(buffer_ + remaining, IStreamBufferSize - remaining))
182 {
183 // In case the buffer is only partially filled, fill the remainder with
184 // EOF characters.
185 remaining += is_.gcount();
186 std::fill_n(buffer_ + remaining, IStreamBufferSize - remaining, EOF);
187 }
188
189 c_ = buffer_;
190 }
191 }
192
193private:
195 void buffer()
196 {
197 if (!is_.read(buffer_, IStreamBufferSize))
198 {
199 // In case the buffer is only partially filled, fill the remainder with
200 // EOF characters.
201 auto remaining = is_.gcount();
202 std::fill_n(buffer_ + remaining, IStreamBufferSize - remaining, EOF);
203 }
204
205 c_ = buffer_;
206 }
207
210 inline void readCharacter()
211 {
212 if (c_++ == eob_) { buffer(); }
213 }
214
221 const char *c_{buffer_ + IStreamBufferSize};
223 const char *eob_{buffer_ + IStreamBufferSize};
224
226 std::istream &is_;
227
229 alignas(64) char buffer_[IStreamBufferSize] = {};
230};

111template<typename T>
112constexpr bool isSame(PlyDataType type)
113{
114 switch (type)
115 {
116 case PlyDataType::Char:
117 return std::is_same<T, char>::value;
118 case PlyDataType::UChar:
119 return std::is_same<T, unsigned char>::value;
120 case PlyDataType::Short:
121 return std::is_same<T, short>::value;
122 case PlyDataType::UShort:
123 return std::is_same<T, unsigned short>::value;
124 case PlyDataType::Int:
125 return std::is_same<T, int>::value;
126 case PlyDataType::UInt:
127 return std::is_same<T, unsigned int>::value;
128 case PlyDataType::Float:
129 return std::is_same<T, float>::value;
130 case PlyDataType::Double:
131 return std::is_same<T, double>::value;
132 }
133
134 return false;
135}

280template<typename T, typename U, typename... Ts>
281constexpr bool isPacked(uintptr_t offset = 0)
282{
283 return ((offset + alignof(T)) % alignof(T)) == 0 && isPacked<U, Ts...>(offset + sizeOf<T>());

307template<typename T>
308struct IsMemcpyable
309{
310 bool operator()(const PlyPropertyConstIterator first, const PlyPropertyConstIterator last) const
311 {
312 return first < last && isSame<T>(first->type());
313 }
314};
315
316template<typename T, std::size_t N>
317struct IsMemcpyable<reflect::Array<T, N>>
318{
319 bool operator()(const PlyPropertyConstIterator, const PlyPropertyConstIterator) const { return false; }
320};
322
323template<typename T, std::size_t N>
324struct IsMemcpyable<reflect::Pack<T, N>>
325{
326 bool operator()(const PlyPropertyConstIterator first, const PlyPropertyConstIterator last) const
327 {
328 return first + N <= last &&
329 std::all_of(first, first + N, [](const PlyProperty &p) { return isSame<T>(p.type()); });
330 }
331};

338template<typename T>
339bool isMemcpyable(const PlyPropertyConstIterator first, const PlyPropertyConstIterator last)
340{
341 return first + detail::numProperties<T>() == last && IsMemcpyable<T>{}(first, last);
342}

344template<typename T, typename U, typename... Ts>
345bool isMemcpyable(const PlyPropertyConstIterator first, const PlyPropertyConstIterator last)
346{
347 return IsMemcpyable<T>{}(first, last) && isMemcpyable<U, Ts...>(first + detail::numProperties<T>(), last);
348}

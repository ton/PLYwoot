32enum class PlyDataType { Char, UChar, Short, UShort, Int, UInt, Float, Double };

34inline std::ostream &operator<<(std::ostream &os, PlyDataType dataType)
35{
36 switch (dataType)
37 {
38 case PlyDataType::Char:
39 return os << "char";
40 case PlyDataType::UChar:
41 return os << "uchar";
42 case PlyDataType::Short:
43 return os << "short";
44 case PlyDataType::UShort:
45 return os << "ushort";
46 case PlyDataType::Int:
47 return os << "int";
48 case PlyDataType::UInt:
49 return os << "uint";
50 case PlyDataType::Float:
51 return os << "float";
52 case PlyDataType::Double:
53 return os << "double";
54 }
55
56 return os;
57}

131 std::pair<PlyProperty, bool> property(const std::string &propertyName) const
132 {
133 const auto it = std::find_if(properties_.begin(), properties_.end(), [&](const PlyProperty &p) {
134 return p.name() == propertyName;
135 });
136 return it != properties_.end() ? std::pair<PlyProperty, bool>{*it, true}
137 : std::pair<PlyProperty, bool>{{}, false};
138 }
139
140 template<typename... Args>
141 PlyProperty &addProperty(Args &&...args)
142 {
143 properties_.emplace_back(std::forward<Args>(args)...);
144 return properties_.back();
145 }
146
147 inline friend bool operator==(const PlyElement &x, const PlyElement &y)
148 {
149 return x.size_ == y.size_ && x.name_ == y.name_ && x.properties_ == y.properties_;
150 }
151 inline friend bool operator!=(const PlyElement &x, const PlyElement &y) { return !(x == y); }
152
153private:
154 std::string name_;
155 std::size_t size_;
156 std::vector<PlyProperty> properties_;
157};

49template<typename T>
50T byte_swap(T t)
51{
52 unsigned char *bytes = reinterpret_cast<unsigned char *>(&t);
53
54 if constexpr (sizeof(T) == 4)
55 {
56 std::swap(bytes[0], bytes[3]);
57 std::swap(bytes[1], bytes[2]);
58 }
59 else if constexpr (sizeof(T) == 8)
60 {
61 std::swap(bytes[0], bytes[7]);
62 std::swap(bytes[1], bytes[6]);
63 std::swap(bytes[2], bytes[5]);
64 std::swap(bytes[3], bytes[4]);
65 }
66
67 return t;
68}

76template<typename T>
77T htobe(T t)
78{
79 if constexpr (sizeof(T) == 1) { return t; }
80 else if constexpr (std::is_integral_v<T> && sizeof(T) == 2) { return static_cast<T>(htobe16(t)); }
81 else if constexpr (std::is_integral_v<T> && sizeof(T) == 4) { return static_cast<T>(htobe32(t)); }
82 else if constexpr (std::is_integral_v<T> && sizeof(T) == 8) { return static_cast<T>(htobe64(t)); }
83 else if constexpr (std::is_floating_point_v<T>) { return byte_swap(t); }
84}

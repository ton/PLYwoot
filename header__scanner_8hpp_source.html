85class HeaderScanner

87public:
90 HeaderScanner(std::istream &is) : is_{is}
91 {
92 if (!is) { throw InvalidInputStream{}; }
93
94 std::string line;
95 while (bool(std::getline(is, line)) && line != endHeaderToken)
96 {
97 buffer_.append(line);
98 buffer_.push_back('\n');
99 }
100
101 if (line == endHeaderToken)
102 {
103 buffer_.append(line);
104 buffer_.push_back('\n');
105 }
106 else { buffer_.push_back(EOF); }
107
108 // Initialize the read head to the start of the buffered data.
109 c_ = buffer_.data();
110 }
111
113 enum class Token {
114 Unknown = 0,
115 Ascii,
116 BinaryBigEndian,
117 BinaryLittleEndian,
118 Char,
119 Comment,
120 Double,
121 Element,
122 EndHeader,
123 Eof,
124 Float,
125 FloatingPointNumber,
126 Format,
127 Identifier,
128 Int,
129 List,
130 MagicNumber,
131 Number,

134 UChar,

140 Token nextToken() noexcept
141 {
142 // Ignore all whitespace, read upto the first non-whitespace character.
143 const char *last = buffer_.data() + buffer_.size();
144 while (c_ < last && 0 <= *c_ && *c_ <= 0x20)
145 {
146 if (*c_ == '\n') ++line_;
147 ++c_;
148 }
149
150 // Read an identifier. After an identifier is read, the read head is
151 // positioned at the start of the next token or whitespace.
152 const char *tokenStart = c_;
153 while (!isTokenDelimiter[(unsigned char)*c_]) { c_++; }
154 tokenString_ = std::string_view(tokenStart, c_ - tokenStart);
155
156 // In case the identifier is one of the reserved keywords, handle it as
157 // such. Use first character for quick comparison.
158 switch (tokenString_.front())
159 {
160 case 'a':
161 token_ = !tokenString_.compare("ascii") ? Token::Ascii : Token::Identifier;
162 break;
163 case 'b':
164 if (!tokenString_.compare("binary_big_endian"))
165 token_ = Token::BinaryBigEndian;
166 else if (!tokenString_.compare("binary_little_endian"))
167 token_ = Token::BinaryLittleEndian;
168 else
169 token_ = Token::Identifier;
170 break;
171 case 'c':
172 if (!tokenString_.compare("char")) { token_ = Token::Char; }
173 else if (!tokenString_.compare("comment"))
174 {
175 token_ = Token::Comment;
176 readComment();
177 }
178 else { token_ = Token::Identifier; }
179 break;
180 case 'd':
181 token_ = (!tokenString_.compare("double") ? Token::Double : Token::Identifier);
182 break;
183 case 'e':
184 if (!tokenString_.compare("element"))
185 token_ = Token::Element;
186 else if (!tokenString_.compare("end_header"))
187 token_ = Token::EndHeader;
188 else
189 token_ = Token::Identifier;
190 break;
191 case 'f':
192 if (!tokenString_.compare("format"))
193 token_ = Token::Format;
194 else if (!tokenString_.compare("float") || !tokenString_.compare("float32"))
195 token_ = Token::Float;
196 else if (!tokenString_.compare("float64"))
197 token_ = Token::Double;
198 else
199 token_ = Token::Identifier;
200 break;
201 case 'l':
202 token_ = (!tokenString_.compare("list") ? Token::List : Token::Identifier);
203 break;
204 case 'i':
205 if (!tokenString_.compare("int") || !tokenString_.compare("int32"))
206 token_ = Token::Int;
207 else if (!tokenString_.compare("int8"))
208 token_ = Token::Char;
209 else if (!tokenString_.compare("int16"))
210 token_ = Token::Short;
211 else
212 token_ = Token::Identifier;
213 break;
214 case 'p':
215 if (!tokenString_.compare("ply"))
216 token_ = Token::MagicNumber;
217 else if (!tokenString_.compare("property"))
218 token_ = Token::Property;
219 else
220 token_ = Token::Identifier;
221 break;
222 case 's':
223 token_ = (!tokenString_.compare("short") ? Token::Short : Token::Identifier);
224 break;
225 case 'u':
226 if (!tokenString_.compare("uint8") || !tokenString_.compare("uchar"))
227 token_ = Token::UChar;
228 else if (!tokenString_.compare("uint16") || !tokenString_.compare("ushort"))
229 token_ = Token::UShort;
230 else if (!tokenString_.compare("uint32") || !tokenString_.compare("uint"))
231 token_ = Token::UInt;
232 else
233 token_ = Token::Identifier;
234 break;
235 case '-':
236 case '+':
237 case '.':
238 case '0':
239 case '1':
240 case '2':
241 case '3':
242 case '4':
243 case '5':
244 case '6':
245 case '7':
246 case '8':
247 case '9':
248 // TODO(ton): scientific notation for floating point numbers?
249 token_ = (tokenString_.find('.') != std::string::npos) ? Token::FloatingPointNumber : Token::Number;
250 break;
251 case EOF:
252 token_ = Token::Eof;
253 break;
254 default:
255 token_ = Token::Identifier;
256 break;
257 }
258
259 return token_;
260 }

263 static constexpr bool isKeyword(Token token)
264 {
265 switch (token)
266 {
267 case Token::Ascii:
268 case Token::BinaryBigEndian:
269 case Token::BinaryLittleEndian:
270 case Token::Char:
271 case Token::Double:
272 case Token::Element:
273 case Token::EndHeader:
274 case Token::Float:
275 case Token::Format:
276 case Token::Int:
277 case Token::List:
278 case Token::Property:
279 case Token::Short:
280 case Token::UChar:
281 case Token::UInt:
282 case Token::UShort:
283 return true;
284 default:
285 break;
286 }
287
288 return false;
289 }

308private:
311 void readComment()
312 {
313 // Skip spaces and tabs and the first non-whitespace character.
314 const char *end = buffer_.data() + buffer_.size();
315 while (c_ < end && (*c_ == ' ' || *c_ == '\t')) { ++c_; }
316
317 const std::size_t remainingBytes = buffer_.size() - (c_ - buffer_.data());
318 const char *last = static_cast<const char *>(::memchr(c_, '\n', remainingBytes));
319 if (last != nullptr)
320 {
321 tokenString_ = std::string_view(c_, last - c_);
322 c_ = last;
323 }
324 }

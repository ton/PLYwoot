155class OStream
156{
157public:
159 OStream(PlyFormat format) : format_{format} {}
165 OStream(PlyFormat format, std::vector<Comment> comments) : format_{format}, comments_{std::move(comments)}
166 {
167 // Ensure comments are sorted on line number (ascending).
168 std::stable_sort(comments_.begin(), comments_.end(), [](const Comment &x, const Comment &y) {
169 return x.line < y.line;
170 });
171 }
172
175 template<typename... Ts>
176 void add(const PlyElement &element, const reflect::Layout<Ts...> layout)
177 {
178 // Note; create a copy of the element that specifies as size the number of
179 // items in the input layout.
180 PlyElement layoutElement{element.name(), layout.size(), element.properties()};
181
182 elementWriteClosures_.emplace_back(
183 std::move(layoutElement),
184 [this, layout](detail::WriterVariant &writer, const PlyElement &e) { writer.write(e, layout); });
185 }
186
190 void add(const PlyElementData &elementData)
191 {
192 // TODO(ton): once we upgrade to C++17, move capture element data in the
193 // lambda below.
194 const std::uint8_t *src = elementData.data();
195 const std::size_t alignment = elementData.alignment();
196
197 elementWriteClosures_.emplace_back(
198 elementData.element(), [this, src, alignment](detail::WriterVariant &writer, const PlyElement &e) {
199 writer.write(e, src, alignment);
200 });
201 }
202
205 void write(std::ostream &os) const
206 {
207 writeHeader(os);
208
209 detail::WriterVariant writer{os, format_};
210 for (const auto &elementClosurePair : elementWriteClosures_)
211 {
212 const PlyElement &element{elementClosurePair.first};
213 const auto &writeFn = elementClosurePair.second;
214 writeFn(writer, element);
215 }
216 }
217
218private:
221 void writeHeader(std::ostream &os) const
222 {
223 os << "ply\n";
224
225 switch (format_)
226 {
227 case PlyFormat::Ascii:
228 os << "format ascii 1.0\n";
229 break;
230 case PlyFormat::BinaryBigEndian:
231 os << "format binary_big_endian 1.0\n";
232 break;
233 case PlyFormat::BinaryLittleEndian:
234 os << "format binary_little_endian 1.0\n";
235 break;
236 }
237
238 // Maintain line number to be able to serialize comments at the right
239 // location in the header. Comments may only occur after the 'ply' magic
240 // 'number', and the format specification. Ideally this state should be
241 // limited to the scope of the closure, but that is not supported in C++11.
242 std::uint32_t line = 2;
243 auto first = comments_.begin();
244 const auto last = comments_.end();
245 const auto maybeWriteComments = [&]() {
246 while (first != last && first->line == line++)
247 {
248 if (first->text.empty()) { os << "comment\n"; }
249 else { os << "comment " << first->text << '\n'; }
250 ++first;
251 }
252 };
253
254 for (const auto &elementClosurePair : elementWriteClosures_)
255 {
256 maybeWriteComments();
257 const PlyElement &element{elementClosurePair.first};
258 os << "element " << element.name() << ' ' << element.size() << '\n';
259
260 for (const PlyProperty &property : element.properties())
261 {
262 maybeWriteComments();
263 if (property.isList())
264 {
265 os << "property list " << property.sizeType() << ' ' << property.type() << ' ' << property.name()
266 << '\n';
267 }
268 else { os << "property " << property.type() << ' ' << property.name() << '\n'; }
269 }
270 }
271
272 maybeWriteComments();
273 os << "end_header\n";
274 }
275
276 using ElementWriteClosure = std::function<void(detail::WriterVariant &, const PlyElement &)>;
277
279 std::vector<std::pair<PlyElement, ElementWriteClosure>> elementWriteClosures_;
281 PlyFormat format_;
284 std::vector<Comment> comments_;
285};

289inline void convert(std::istream &is, std::ostream &os, PlyFormat format)
290{
291 IStream plyIs{is};
292 OStream plyOs{format};
293
294 // Note; preserve element data until the write() function below finishes.
295 std::vector<PlyElementData> elementsData;
296
297 while (plyIs.hasElement())
298 {
299 const plywoot::PlyElement element = plyIs.element();
300 elementsData.emplace_back(plyIs.readElement());
301 plyOs.add(elementsData.back());
302 }
303
304 plyOs.write(os);
305}

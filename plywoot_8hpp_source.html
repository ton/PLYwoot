212 template<typename... Ts>
213 void add(const PlyElement &element, const reflect::Layout<Ts...> layout)
214 {
215 // Note; create a copy of the element that specifies as size the number of
216 // items in the input layout.
217 PlyElement layoutElement{element.name(), layout.size(), element.properties()};
218
219 elementWriteClosures_.emplace_back(
220 std::move(layoutElement),
221 [this, layout](detail::WriterVariant &writer, const PlyElement &e) { writer.write(e, layout); });
222 }
223
229 void add(const PlyElementData &elementData)
230 {
231 // TODO(ton): once we upgrade to C++17, move capture element data in the
232 // lambda below.
233 const std::uint8_t *src = elementData.data();
234 const std::size_t alignment = elementData.alignment();
235
236 elementWriteClosures_.emplace_back(
237 elementData.element(), [this, src, alignment](detail::WriterVariant &writer, const PlyElement &e) {
238 writer.write(e, src, alignment);
239 });
240 }
241
246 void write(std::ostream &os) const
247 {
248 writeHeader(os);
249
250 detail::WriterVariant writer{os, format_};
251 for (const auto &elementClosurePair : elementWriteClosures_)
252 {
253 const PlyElement &element{elementClosurePair.first};
254 const auto &writeFn = elementClosurePair.second;
255 writeFn(writer, element);
256 }
257 }
258
259private:
264 void writeHeader(std::ostream &os) const
265 {
266 os << "ply\n";
267
268 switch (format_)
269 {
270 case PlyFormat::Ascii:
271 os << "format ascii 1.0\n";
272 break;
273 case PlyFormat::BinaryBigEndian:
274 os << "format binary_big_endian 1.0\n";
275 break;
276 case PlyFormat::BinaryLittleEndian:
277 os << "format binary_little_endian 1.0\n";
278 break;
279 }
280
281 // Maintain line number to be able to serialize comments at the right
282 // location in the header. Comments may only occur after the 'ply' magic
283 // 'number', and the format specification. Ideally this state should be
284 // limited to the scope of the closure, but that is not supported in C++11.
285 std::uint32_t line = 2;
286 auto first = comments_.begin();
287 const auto last = comments_.end();
288 const auto maybeWriteComments = [&]() {
289 while (first != last && first->line == line++)
290 {
291 if (first->text.empty()) { os << "comment\n"; }
292 else { os << "comment " << first->text << '\n'; }
293 ++first;
294 }
295 };
296
297 for (const auto &elementClosurePair : elementWriteClosures_)
298 {
299 maybeWriteComments();
300 const PlyElement &element{elementClosurePair.first};
301 os << "element " << element.name() << ' ' << element.size() << '\n';
302
303 for (const PlyProperty &property : element.properties())
304 {
305 maybeWriteComments();
306 if (property.isList())
307 {
308 os << "property list " << property.sizeType() << ' ' << property.type() << ' ' << property.name()
309 << '\n';
310 }
311 else { os << "property " << property.type() << ' ' << property.name() << '\n'; }
312 }
313 }
314
315 maybeWriteComments();
316 os << "end_header\n";
317 }
318
319 using ElementWriteClosure = std::function<void(detail::WriterVariant &, const PlyElement &)>;
320
322 std::vector<std::pair<PlyElement, ElementWriteClosure>> elementWriteClosures_;
324 PlyFormat format_;
327 std::vector<Comment> comments_;
328};

336inline void convert(std::istream &is, std::ostream &os, PlyFormat format)
337{
338 IStream plyIs{is};
339 OStream plyOs{format};
340
341 // Note; preserve element data until the write() function below finishes.
342 std::vector<PlyElementData> elementsData;
343
344 while (plyIs.hasElement())
345 {
346 const plywoot::PlyElement element = plyIs.element();
347 elementsData.emplace_back(plyIs.readElement());
348 plyOs.add(elementsData.back());
349 }
350
351 plyOs.write(os);
352}
